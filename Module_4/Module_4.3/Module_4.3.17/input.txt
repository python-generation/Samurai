from typing import Any

class CustomList:
    def __init__(self):
        self._data = []

    def append(self, value: Any) -> None:
        self._data.append(value)

    def pop(self, index: int = -1) -> Any:
        return self._data.pop(index)

    def insert(self, index: int, value: Any) -> None:
        self._data.insert(index, value)

    def __repr__(self):
        return f'CustomList({self._data})'

custom_list = CustomList()
custom_list.append('a')
custom_list.insert(0, 'b')
popped = custom_list.pop(1)

print(f'{custom_list}, popped: {popped}')
print(custom_list._data == ['b'] and popped == 'a')
print(isinstance(custom_list, SupportsAppendPopIns))
print(issubclass(CustomList, SupportsAppendPopIns))
