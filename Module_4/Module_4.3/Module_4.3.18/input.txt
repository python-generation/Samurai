# TEST_5:
class NetworkConnection:
    def __init__(self, host, port):
        self.host = host
        self.port = port
        self.connected = True
        print(f'Connected to {self.host}:{self.port}')

    def close(self) -> None:
        if self.connected:
            print(f'Closing connection to {self.host}:{self.port}')
            self.connected = False

    def __repr__(self):
        return f'NetworkConnection(host="{self.host}", port={self.port}, connected={self.connected})'

conn1 = NetworkConnection('localhost', 8080)
conn2 = NetworkConnection('example.com', 443)
connections = [conn1, conn2]

close_all(connections)

print(conn1)
print(conn2)
print(conn1.connected is False and conn2.connected is False)
print(isinstance(conn1, SupportsClose))
print(issubclass(NetworkConnection, SupportsClose))

# TEST_7:
class AnotherResource:
    def close(self) -> None:
        self.closed = True
        print('AnotherResource closed.')
        return None  # Явное возвращение None

    def __init__(self):
        self.closed = False
    def __repr__(self):
        return f'AnotherResource(closed={self.closed})'

temp_file = open('temp_file.txt', 'w')
another_resource = AnotherResource()
mixed_closable = (temp_file, another_resource) # Используем кортеж

close_all(mixed_closable)

print(temp_file.closed)
print(another_resource)
print(temp_file.closed is True and another_resource.closed is True)
print(isinstance(temp_file, SupportsClose))
print(isinstance(another_resource, SupportsClose))
