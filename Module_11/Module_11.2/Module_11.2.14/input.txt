class Bee(UpperAttr):
    geek = 'pygen'                  # публичный атрибут, должен быть преобразован
    _geek = 'pygen'                 # защищенный атрибут, должен быть преобразован
    __geek = 'pygen'                # приватный атрибут, не должен быть преобразован

    def get_private(self):
        return self.__geek

bee = Bee()
print(bee.GEEK)
print(bee._GEEK)
print(bee.get_private())
