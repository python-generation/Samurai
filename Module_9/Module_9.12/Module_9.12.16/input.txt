conn = sqlite3.connect(':memory:')

factory = FilteringFactory(skip_empty=True)
conn.row_factory = factory
cursor = conn.cursor()

cursor.executescript('''
DROP TABLE IF EXISTS Songs;
CREATE TABLE Songs
(
    id        INTEGER PRIMARY KEY,
    trackname TEXT,
    artist    TEXT
);

INSERT INTO Songs (id, trackname, artist)
VALUES (1, 'Crazy on You', 'Heart'),
       (2, '', 'The Sounds'),
       (3, 'Spent the Day in Bed', NULL),
       (4, 'Bigmouth Strikes Again', 'The Smiths'),
       (5, 'Everyday is Like Sunday', 'Morrissey');
''')

conn.commit()

cursor.execute('SELECT * FROM Songs;')

for row in cursor.fetchall():
    print(row)

for rec in factory.records:
    print(rec)

cursor.close()
conn.close()
